def to_display
  Inkblot::Components::SimpleText.new do |st|
    st.fullscreen = true
    st.text = @chronometer[:ads_blocked_today].to_s
    st.size = 40
  end
end

def read_chronometer
  @chronometer = JSON.parse(`pihole chronometer -j`)
                     .transform_keys(&:to_sym)
end
