def computed
  dta = OpenStruct.new

  get_height(dta)
  get_width(dta)

  dta.source = case img_src
               when :url
                 options[:url]
               when :path
                 File.absolute_path(options[:path])
               when :file
                 bs = +"" << 'data:image/'
                 bs << File.extname(options[:file]).gsub(/[^a-z]/i, '')
                 bs << ';base64,'

                 binary = File.read(options[:file].path)
                 bs << Base64.encode64(binary).gsub("\n", '')
                 bs
               end

  dta.to_h
end
