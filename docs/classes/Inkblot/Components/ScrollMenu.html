def button_actions
  case state
  when :scroll
    [
      -> { self.state = :select },
      -> { self.state = :canceled },
      -> { self.prev_page },
      -> { self.next_page }
    ]
  when :select
    Array.new(4) do |i|
      lambda do
        ch = choice(i)
        return unless ch

        @answer = ch
        self.state = :answered
      end
    end
  else
    []
  end
end

pad_list ()

Sets up pagination by creating a new array that includes the list start & end, and slicing it into fours. Creates content for relevant states with different icons.

[show source] 
# File lib/inkblot/components/scroll_menu.rb, line 86
def pad_list
  pg = [LIST_START]

  pg += items

  pg << LIST_END

  dx = items.each_with_index.to_h

  scroll_content(pg, dx) && select_content(pg, dx)

  pg.each_slice(4).count
end

def scroll_content(pages, indx)
  content_for_state(:scroll) do
    pages.each_slice(4).map do |page|
      IconPane.new do |pane|
        pane.icons = :select
        pane.div_height = pane.div_width = :full
        pane.choices = page.map { |g| indx[g] }
        pane.frame_contents = TableList.new do |lst|
          lst.items = page.map(&:to_s)
        end
      end
    end
  end
end

def select_content(pages, indx)
  content_for_state(:select) do
    pages.each_slice(4).map do |page|
      IconPane.new do |pane|
        pane.div_height = pane.div_width = :full
        pane.choices = page.map { |g| indx[g] }
        pane.icons = pane.choices.map { |ch| ch.nil? ? :times : :rarr }
        pane.frame_contents = TableList.new do |lst|
          lst.items = page.map(&:to_s)
        end
      end
    end
  end
end
