    options[:buttons] = Array.new(4) { |x| "Key #{x}" }
  end

def computed
  dta = OpenStruct.new
  
  get_height(dta)
  get_width(dta)
  
  dta.border_size = options.fetch(:border_size, 0)

  dta.to_h
end
